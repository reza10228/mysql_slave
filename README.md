#Create user on docker mysql master

CREATE USER 'USER'@'%' IDENTIFIED WITH mysql_native_password BY 'PASSWORD';
GRANT ALL PRIVILEGES ON *.* TO 'USER'@'%';
FLUSH PRIVILEGES;
show grants for 'USER'@'%';


#execute command on mysql master

SHOW MASTER STATUS;
